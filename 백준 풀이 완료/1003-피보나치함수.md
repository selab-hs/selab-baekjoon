* 동적 프로그래밍의 Bottom-Up 방식 사용
```java
import java.util.*;

public class Main {

    static Map<Integer, ZeroAndOne> memorization = new HashMap<>();

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String chance = scanner.nextLine();
        int[] cases = new int[Integer.parseInt(chance)];
        for(int i=0;i< Integer.parseInt(chance);i++){
            cases[i] = Integer.parseInt(scanner.nextLine());
        }
        OptionalInt max = Arrays.stream(cases).max();
        calculate(max);

        for(int i=0;i<cases.length;i++){
            System.out.println(memorization.get(cases[i]));
        }

    }

    private static void calculate(OptionalInt max) {
        int m = max.getAsInt();
        memorization.put(0,new ZeroAndOne(1,0));
        memorization.put(1,new ZeroAndOne(0,1));
        memorization.put(2,new ZeroAndOne(1,1));
        for (int i = 2; i <= m; i++) {
            memorization.put(
                    i, addData(
                            memorization.get(i-1),
                            memorization.get(i-2)
                    )
            );
        }
    }

    private static ZeroAndOne addData(ZeroAndOne a, ZeroAndOne b){
        return new ZeroAndOne(
                a.getZero() + b.getZero(),
                a.getOne() + b.getOne()
        );
    }
}

class ZeroAndOne {
    private final int zero;
    private final int one;

    public ZeroAndOne(int zero, int one) {
        this.zero = zero;
        this.one = one;
    }

    public int getOne() {
        return one;
    }

    public int getZero() {
        return zero;
    }

    @Override
    public String toString() {
        return zero + " " + one;
    }
}
```
